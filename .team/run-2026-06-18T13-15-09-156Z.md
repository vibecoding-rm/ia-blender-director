```diff
+"""Regression tests for server.run_plan_pipeline_async.
+
+The P1 bug was a NameError raised while running the plan pipeline coroutine.
+These tests run the coroutine end-to-end with all external/LLM and rendering
+calls mocked out (no network, no Blender, no ComfyUI) and assert that:
+
+  * the coroutine completes without raising (in particular, no NameError), and
+  * the plan finishes in a success state ("completed"), never "failed".
+"""
+
+import asyncio
+import tempfile
+from pathlib import Path
+from unittest import TestCase
+from unittest.mock import AsyncMock, patch
+
+from ai_blender_director.jobs import RenderJob
+
+
+def _last_status(upsert_mock):
+    """Return the status of the most recent _upsert_plan(plan_id, data) call."""
+    assert upsert_mock.await_args_list, "_upsert_plan was never awaited"
+    _plan_id, data = upsert_mock.await_args_list[-1].args
+    return data.get("status")
+
+
+class TestRunPlanPipelineAsync(TestCase):
+    def test_no_jobs_completes_without_nameerror(self):
+        """With an empty shot list the coroutine takes the no-video branch and
+        must still settle on a 'completed' status (and not raise NameError)."""
+        from ai_blender_director import server
+
+        upsert = AsyncMock()
+        with patch.object(server, "_upsert_plan", upsert), \
+                patch.object(server.broadcaster, "schedule_cleanup"):
+            # asyncio.run guarantees a real running loop (schedule_cleanup /
+            # broadcaster rely on asyncio.create_task).
+            asyncio.run(server.run_plan_pipeline_async(
+                plan_id="plan_test_empty",
+                jobs=[],
+                workflow="default",
+                fps=24,
+                hook_title="Dummy Hook",
+                narration_text="Esta es una narracion de prueba.",
+            ))
+
+        status = _last_status(upsert)
+        self.assertEqual(status, "completed")
+        self.assertNotEqual(status, "failed")
+
+    def test_with_mocked_shot_does_not_raise(self):
+        """Exercise the per-shot loop (the code path that previously hit the
+        NameError) with the inner render pipeline fully mocked."""
+        from ai_blender_director import server
+
+        with tempfile.TemporaryDirectory() as tmp:
+            job_dir = Path(tmp) / "job_0"
+            job_dir.mkdir(parents=True)
+            job = RenderJob(
+                job_id="job_0",
+                job_dir=job_dir.resolve(),
+                source_shot=(job_dir / "shot.json").resolve(),
+                job_shot=(job_dir / "shot.json").resolve(),
+                output_root=Path(tmp).resolve(),
+                profile="default",
+            )
+
+            upsert = AsyncMock()
+            # _run_pipeline_inner is async and must not touch Blender/ComfyUI.
+            inner = AsyncMock()
+            with patch.object(server, "_upsert_plan", upsert), \
+                    patch.object(server, "_run_pipeline_inner", inner), \
+                    patch.object(server.broadcaster, "schedule_cleanup"):
+                asyncio.run(server.run_plan_pipeline_async(
+                    plan_id="plan_test_shot",
+                    jobs=[job],
+                    workflow="default",
+                    fps=24,
+                    hook_title="Dummy Hook",
+                    narration_text="Narracion dummy para el plan.",
+                ))
+
+            # The mocked inner pipeline was invoked for the single shot...
+            inner.assert_awaited()
+            # ...and since it produced no final_render.mp4, the plan still
+            # completes successfully via the no-video branch.
+            self.assertEqual(_last_status(upsert), "completed")
```
